‚ÄúHow many of our customers are Cloud, On-Prem, or Hybrid?‚Äù (conservative rules)

-- Part 2 (per-account flags)
WITH latest_deploy AS (
  SELECT account_id, MAX(active_from) AS active_from
  FROM deployments
  WHERE active_to IS NULL
  GROUP BY account_id
)
SELECT a.account_id,
       CASE WHEN d.provider IS NOT NULL THEN 1 ELSE 0 END AS has_cloud_flag,
       CASE WHEN t.source = 'onprem' THEN 1 ELSE 0 END AS has_onprem_flag
FROM accounts a
LEFT JOIN latest_deploy ld USING (account_id)
LEFT JOIN deployments d USING (account_id, active_from)
LEFT JOIN telemetry t ON t.account_id = a.account_id;
